Let's make a lottery contract!*

*Shout-out to Stephen Grider for his online course!

Lottery Contract in code (Solidity)


contract Lottery {
  address public manager;
  address[] public players;
  event newPlayer();
  event winnerPicked();
  
  function constructor() public {
    manager = msg.sender;
  }
  
  function enter() public payable {
    require(msg.value > .01 ether);
    players.push(msg.sender);
  }
  
  function random() private view returns (uint) {
    return uint(sha3(block.difficulty, now, players));
  }
  
  function pickWinner() public {
    require(msg.sender == manager);
    uint index = random() % players.length;
    players[index].transfer(this.balance);
    players = new address[](0);
  }
  
  function getPlayers() public view returns (address[]) {
    return players;
  }
}

How do we talk to our Smart Contract?

web3.js

web3.js

  • JavaScript RPC client
  • Provides access to wallet accounts
  • Sending transactions
  • Creates Contract instances from ABI
  • Utilities
import Web3 from 'web3';

const web3 = new Web3(provider) // HD Wallet or Metamask
const accounts = await web3.eth.getAccounts();
const lottery = new web3.eth.Contract(abi, '0x...');

await lottery.methods.enter().send({
  from: accounts[0],
  value: web3.utils.toWei('1', 'ether')
});

const players = await lottery.methods.getPlayers().call();

          

@NgModule({
  providers: [
    { 
      provide: Web3, useFactory: () => { return new Web3(provider) }
    },
    { 
      provide: Lottery, 
      deps: [Web3],
      useFactory: (web3) => {
        return new web3.eth.Contract(abi, '0x...');
      }
    }
  ]
})
export class AppModule {...}
          
@Component(...)
class AppComponent {
  constructor(private web3: Web3, private lottery: Lottery) {}

  async pickWinner() {
    const accounts = await this.web3.getAccounts();

    await this.lottery.methods.pickWinner().send({
      from: accounts[0];
    });
  }
}
